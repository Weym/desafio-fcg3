Add a `POST /documents/upload` endpoint to the existing documents routes file. If `src/backend/src/features/documents/routes.py` doesn't exist, create it and register in `main.py`.

**Endpoint specification:**
- Path: `POST /api/v1/documents/upload`
- Auth: Staff only (JWT Bearer with staff role check)
- Content-Type: `multipart/form-data`
- Accepts: single file field named `file`
- Validates: file type must be PDF, PNG, or JPG; max size 10MB
- Storage: Save to `uploads/documents/` directory (local filesystem for MVP)
- Returns: `{"url": "/uploads/documents/{uuid}_{filename}", "filename": "{original_filename}"}`

**Implementation:**
```python
import os
import uuid
from fastapi import APIRouter, UploadFile, File, HTTPException, Depends

# If the file already has a router, add this route to it
# Otherwise create the router

UPLOAD_DIR = "uploads/documents"
MAX_FILE_SIZE = 10 * 1024 * 1024  # 10MB
ALLOWED_EXTENSIONS = {".pdf", ".png", ".jpg", ".jpeg"}

@router.post("/documents/upload")
async def upload_document(
    file: UploadFile = File(...),
    # current_user: UserModel = Depends(get_current_staff_user)  # Staff auth
):
    # Validate extension
    ext = os.path.splitext(file.filename or "")[1].lower()
    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(status_code=400, detail={
            "error": {"code": "INVALID_FILE_TYPE", "message": f"Allowed types: PDF, PNG, JPG. Got: {ext}"}
        })
    
    # Read and validate size
    content = await file.read()
    if len(content) > MAX_FILE_SIZE:
        raise HTTPException(status_code=400, detail={
            "error": {"code": "FILE_TOO_LARGE", "message": "Maximum file size is 10MB"}
        })
    
    # Save file
    os.makedirs(UPLOAD_DIR, exist_ok=True)
    file_id = str(uuid.uuid4())
    safe_filename = f"{file_id}_{file.filename}"
    file_path = os.path.join(UPLOAD_DIR, safe_filename)
    
    with open(file_path, "wb") as f:
        f.write(content)
    
    return {
        "url": f"/uploads/documents/{safe_filename}",
        "filename": file.filename,
    }
```
