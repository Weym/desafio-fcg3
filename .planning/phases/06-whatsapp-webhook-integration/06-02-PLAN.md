**1. Create Background Task Module (backend/src/features/webhook/background.py):**

This is the critical module that bridges the webhook to the AI service. Every design choice here is driven by specific pitfall mitigations.

```python
import asyncio
import logging
import httpx
from uuid import UUID

from backend.src.infrastructure.database import async_session_maker
from backend.src.infrastructure.whatsapp_client import WhatsAppClient
from backend.src.infrastructure.settings import settings

logger = logging.getLogger(__name__)

# Per-session locks to prevent concurrent processing (D-09, MINOR-3)
_session_locks: dict[str, asyncio.Lock] = {}

FALLBACK_MESSAGE = "Desculpe, estou com dificuldades tecnicas. Tente novamente em alguns minutos."


def _handle_task_result(task: asyncio.Task) -> None:
    """Done callback for background tasks. Logs exceptions that would be silently swallowed (CRITICAL-3)."""
    if task.cancelled():
        logger.warning("Background message processing task was cancelled")
        return
    exc = task.exception()
    if exc is not None:
        logger.error(
            "Background task failed with unhandled exception",
            exc_info=exc,
        )


async def process_verified_message(
    session_id: UUID,
    message_text: str,
    phone: str,
    wa_client: WhatsAppClient,
) -> None:
    """
    Process a verified student's text message:
    1. Acquire per-session lock (D-09)
    2. Call AI service with message + session_id
    3. On failure: retry once, then send fallback (D-06)
    4. Save assistant response to chat_messages
    5. Send response via WhatsApp
    
    CRITICAL-4: Opens its OWN DB session — never uses the request-scoped one.
    """
    lock = _session_locks.setdefault(str(session_id), asyncio.Lock())
    
    async with lock:
        agent_response: str | None = None
        
        # Call AI service with one retry on failure (D-06)
        for attempt in range(2):
            try:
                async with httpx.AsyncClient(timeout=httpx.Timeout(50.0, connect=5.0)) as http:
                    response = await http.post(
                        f"{settings.AI_SERVICE_URL}/chat",
                        json={
                            "session_id": str(session_id),
                            "message": message_text,
                        },
                    )
                    if response.status_code == 200:
                        data = response.json()
                        agent_response = data.get("response", FALLBACK_MESSAGE)
                        break
                    else:
                        logger.warning(
                            f"AI service returned {response.status_code} on attempt {attempt+1}"
                        )
            except (httpx.HTTPError, Exception) as e:
                logger.warning(f"AI service call attempt {attempt+1} failed: {e}")
        
        # If both attempts failed, use fallback (D-06)
        if agent_response is None:
            agent_response = FALLBACK_MESSAGE
            logger.error(f"AI service unavailable for session {session_id}, sending fallback")
        
        # Save assistant response to chat_messages (CRITICAL-4: own session)
        async with async_session_maker() as db:
            from backend.src.features.webhook.service import WebhookService
            webhook_service = WebhookService()
            await webhook_service.save_message(
                session_id=session_id,
                role="assistant",
                content=agent_response,
                media_type=None,
                wamid=None,
                db=db,
            )
            await db.commit()
        
        # Send response via WhatsApp (D-07: WhatsApp client already handles retry)
        await wa_client.send_text_message(phone, agent_response)
```

**1. Enhance Session Lifecycle in WebhookService (per D-10, D-13):**

Update `get_or_create_session` to handle the full lifecycle:

```python
async def get_or_create_session(
    self, student_id: UUID, phone: str, db: AsyncSession
) -> ChatSession:
    """
    D-10: Reuse active session per phone number.
    D-13: Closed session → new session, verification_state starts unverified.
    """
    # Look for existing active session
    result = await db.execute(
        select(ChatSession).where(
            ChatSession.whatsapp_phone == phone,
            ChatSession.status == "active",
        )
    )
    session = result.scalar_one_or_none()
    
    if session is not None:
        # Update updated_at for auto-close tracking
        session.updated_at = datetime.utcnow()  # NOTE: requires updated_at column
        await db.commit()
        return session
    
    # No active session — create new one (D-13: starts unverified)
    new_session = ChatSession(
        id=uuid4(),
        student_id=student_id,
        whatsapp_phone=phone,
        status="active",
        verification_state="unverified",
        started_at=datetime.utcnow(),
    )
    db.add(new_session)
    await db.commit()
    await db.refresh(new_session)
    return new_session
```

**2. Create pg_cron Alembic Migration (per D-12):**

This migration:
- Creates the `pg_cron` extension (requires the Docker image to have pg_cron installed — Phase 1 responsibility)
- Adds `updated_at` column to `chat_sessions` if not already present
- Schedules the auto-close job

```python
"""Add pg_cron session auto-close job

Revision ID: xxx
"""
from alembic import op
import sqlalchemy as sa

def upgrade():
    # Add updated_at to chat_sessions if not present (needed for auto-close tracking)
    op.add_column(
        "chat_sessions",
        sa.Column("updated_at", sa.DateTime(), server_default=sa.func.now(), nullable=False),
    )
    
    # Create pg_cron extension
    op.execute("CREATE EXTENSION IF NOT EXISTS pg_cron")
    
    # Schedule session auto-close job per D-12
    # Runs every hour, closes sessions inactive for 24+ hours
    op.execute("""
        SELECT cron.schedule(
            'close-inactive-chat-sessions',
            '0 * * * *',
            $$UPDATE chat_sessions SET status = 'closed', ended_at = NOW() WHERE updated_at < NOW() - INTERVAL '24 hours' AND status = 'active'$$
        )
    """)

def downgrade():
    op.execute("SELECT cron.unschedule('close-inactive-chat-sessions')")
    op.execute("DROP EXTENSION IF EXISTS pg_cron")
    op.drop_column("chat_sessions", "updated_at")
```

IMPORTANT: For this migration to work, Phase 1 must build a custom PostgreSQL Docker image that includes pg_cron (per D-12: "requires custom PostgreSQL Docker image with both pgvector and pg_cron extensions"). The Dockerfile should extend `pgvector/pgvector:pg16` and add:
```dockerfile
RUN apt-get update && apt-get install -y postgresql-16-cron && rm -rf /var/lib/apt/lists/*
```
And add `shared_preload_libraries = 'pg_cron'` to postgresql.conf.

This Phase 6 migration assumes that infrastructure is in place.

| FastAPI → AI Service | Internal HTTP call to langchain-service:8001 |
| Background task → DB | Async session opened after request ends |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-08 | Denial of Service | Background task exception | mitigate | done_callback logs all exceptions (CRITICAL-3). Fallback message sent on failure (D-06). |
| T-06-09 | Tampering | AI service response | accept | Internal network (Docker app-network). AI service is trusted. |
| T-06-10 | Denial of Service | Rapid concurrent messages | mitigate | Per-session asyncio.Lock serializes processing (D-09, MINOR-3). |
| T-06-11 | Information Disclosure | DB session leaked to background | mitigate | Background task opens own session via async_session_maker (CRITICAL-4). |
| T-06-12 | Elevation of Privilege | pg_cron job | accept | Job runs as DB superuser within the container; no external access. Standard pg_cron security model. |
