Implementation for the CustomPainter — draw a stylized alpha (α) symbol:

```dart
import 'package:flutter/material.dart';

/// Theme-adaptive Alpha Connect logo mark.
/// Renders the stylized α symbol as a vector path that works at any size
/// and adapts to light/dark theme automatically.
class AlphaConnectLogo extends StatelessWidget {
  const AlphaConnectLogo({
    super.key,
    required this.size,
    this.color,
  });

  final double size;
  final Color? color;

  @override
  Widget build(BuildContext context) {
    final effectiveColor = color ?? Theme.of(context).colorScheme.primary;
    return CustomPaint(
      size: Size(size, size),
      painter: _AlphaLogoPainter(color: effectiveColor),
    );
  }
}

class _AlphaLogoPainter extends CustomPainter {
  _AlphaLogoPainter({required this.color});

  final Color color;

  @override
  void paint(Canvas canvas, Size size) {
    final paint = Paint()
      ..color = color
      ..style = PaintingStyle.stroke
      ..strokeWidth = size.width * 0.08
      ..strokeCap = StrokeCap.round
      ..strokeJoin = StrokeJoin.round;

    final w = size.width;
    final h = size.height;

    // Draw the α (alpha) symbol — a circular open loop with a tail
    // The path forms: start from right-mid, sweep left around to form the bowl,
    // then cross back right and extend into a descending tail (the "x" flourish)
    
    final path = Path();
    
    // Main oval body of the α — centered, taking ~70% of canvas
    final centerX = w * 0.45;
    final centerY = h * 0.42;
    final radiusX = w * 0.28;
    final radiusY = h * 0.28;
    
    // Start from the right opening (where the tail begins)
    // Arc from top-right, going counter-clockwise around to form the α bowl
    path.moveTo(centerX + radiusX * 0.9, centerY - radiusY * 0.3);
    
    // Upper right curve
    path.cubicTo(
      centerX + radiusX * 0.9, centerY - radiusY * 1.1,
      centerX - radiusX * 0.3, centerY - radiusY * 1.2,
      centerX - radiusX, centerY - radiusY * 0.2,
    );
    
    // Left side down
    path.cubicTo(
      centerX - radiusX * 1.3, centerY + radiusY * 0.8,
      centerX - radiusX * 0.4, centerY + radiusY * 1.3,
      centerX + radiusX * 0.5, centerY + radiusY * 0.9,
    );
    
    // Bottom curve back to right — closing the α loop
    path.cubicTo(
      centerX + radiusX * 1.0, centerY + radiusY * 0.6,
      centerX + radiusX * 1.0, centerY - radiusY * 0.1,
      centerX + radiusX * 0.9, centerY - radiusY * 0.3,
    );
    
    canvas.drawPath(path, paint);
    
    // Draw the tail/flourish extending down-right (the "x" part)
    final tailPath = Path();
    tailPath.moveTo(centerX + radiusX * 0.6, centerY + radiusY * 0.5);
    tailPath.cubicTo(
      centerX + radiusX * 1.2, centerY + radiusY * 1.0,
      w * 0.75, h * 0.7,
      w * 0.82, h * 0.82,
    );
    
    canvas.drawPath(tailPath, paint);
    
    // Small cross stroke for the "x" character
    final crossPath = Path();
    crossPath.moveTo(w * 0.68, h * 0.78);
    crossPath.cubicTo(
      w * 0.72, h * 0.72,
      w * 0.78, h * 0.68,
      w * 0.85, h * 0.62,
    );
    
    canvas.drawPath(crossPath, paint);
  }

  @override
  bool shouldRepaint(covariant _AlphaLogoPainter oldDelegate) {
    return oldDelegate.color != color;
  }
}
```

The exact path control points above are starting points — the executor should adjust the cubic bezier control points as needed to produce a clean α shape that visually matches the original logo mark. The key requirements are:
- Recognizable as the Greek letter alpha (α) with a stylized tail
- Clean at 36px (appbar), 80px (login), and 100px (splash)
- Single color, no fill — stroke only for elegance
- Stroke width proportional to size (8% of width)
