    async def seed_scheduling(session: AsyncSession):
        """Create a resource and scheduling slots for testing."""

            name="Sala de Coordenação",
            resource_type="room",

            location="Bloco 3, Sala 301",

        today = date.today()
        slots = [
            {"date": today + timedelta(days=1), "start": "09:00", "end": "10:00"},
            {"date": today + timedelta(days=1), "start": "10:00", "end": "11:00"},
            {"date": today + timedelta(days=2), "start": "14:00", "end": "15:00"},
            {"date": today + timedelta(days=3), "start": "09:00", "end": "10:00"},
            {"date": today + timedelta(days=3), "start": "10:00", "end": "11:00"},
        ]
        for s in slots:
            from datetime import time as dt_time
            sh, sm = map(int, s["start"].split(":"))
            eh, em = map(int, s["end"].split(":"))
            slot = SchedulingSlot(
                resource_id=resource.id,
                date=s["date"],
                start_time=dt_time(sh, sm),
                end_time=dt_time(eh, em),
                is_available=True,
            )
            session.add(slot)

        await session.commit()
