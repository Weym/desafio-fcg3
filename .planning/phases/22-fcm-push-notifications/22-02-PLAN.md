2. Create `mobile/lib/core/providers/fcm_provider.dart`:
   ```dart
   import 'package:firebase_messaging/firebase_messaging.dart';
   import 'package:riverpod_annotation/riverpod_annotation.dart';
   import '../providers/dio_provider.dart';
   import '../../features/auth/providers/auth_provider.dart';

   part 'fcm_provider.g.dart';

   @Riverpod(keepAlive: true)
   class FcmService extends _$FcmService {
     @override
     void build() {
       // Listen for token refresh (per D-06)
       FirebaseMessaging.instance.onTokenRefresh.listen(_onTokenRefresh);
     }

     /// Request notification permission (per D-26: immediately after first login)
     /// Returns true if permission granted.
     Future<bool> requestPermission() async {
       final settings = await FirebaseMessaging.instance.requestPermission(
         alert: true,
         badge: true,
         sound: true,
       );
       return settings.authorizationStatus == AuthorizationStatus.authorized ||
              settings.authorizationStatus == AuthorizationStatus.provisional;
     }

     /// Register FCM token with backend (per D-01: immediately after login)
     /// Per D-29: Only register if permission was granted.
     Future<void> registerToken(String studentId) async {
       try {
         final granted = await requestPermission();
         if (!granted) return; // Per D-29: no token if denied

         final token = await FirebaseMessaging.instance.getToken();
         if (token == null) return;

         final dio = ref.read(dioProvider);
         await dio.put(
           '/students/$studentId/fcm-token',
           data: {'token': token},
         );
       } catch (e) {
         // Fire-and-forget — don't crash auth flow if FCM registration fails
       }
     }

     /// Unregister FCM token from backend (per D-02: on logout)
     Future<void> unregisterToken(String studentId) async {
       try {
         final token = await FirebaseMessaging.instance.getToken();
         if (token == null) return;

         final dio = ref.read(dioProvider);
         await dio.delete(
           '/students/$studentId/fcm-token',
           data: {'token': token},
         );
       } catch (e) {
         // Best effort — don't block logout
       }
     }

     /// Handle token refresh (per D-06)
     void _onTokenRefresh(String newToken) async {
       // Get current auth state to find student ID
       final authState = ref.read(authProvider);
       if (authState is! AuthAuthenticated) return;
       if (!authState.user.isStudent) return;

       try {
         final dio = ref.read(dioProvider);
         await dio.put(
           '/students/${authState.user.id}/fcm-token',
           data: {'token': newToken},
         );
       } catch (e) {
         // Best effort
       }
     }
   }
   ```

| Flutter app → Firebase SDK | Token obtained from Firebase servers |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-22-06 | Spoofing | Token registration | mitigate | JWT auth required for PUT /students/{id}/fcm-token; IDOR check on student_id path param |
| T-22-07 | Information Disclosure | FCM token in transit | mitigate | All API calls over HTTPS (Dio baseUrl). Token is opaque device identifier, not PII. |
| T-22-08 | Denial of Service | Permission denial | accept | If user denies permission, no token registered (per D-29). App functions normally without push. |

</threat_model>
