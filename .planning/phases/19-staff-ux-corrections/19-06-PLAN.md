2. Create `mobile/lib/features/staff/services/staff_cadastro_service.dart`:
   ```dart
   import '../../../core/network/dio_client.dart';
   import '../models/staff_student_model.dart';

   class StaffCadastroService {
     final DioClient client;
     StaffCadastroService({required this.client});

     Future<List<StaffStudentModel>> getStudents() async {
       final response = await client.get('/students');
       final List<dynamic> data = response.data is List ? response.data : response.data['data'] ?? [];
       return data.map((e) => StaffStudentModel.fromJson(e as Map<String, dynamic>)).toList();
     }

     Future<StaffStudentModel> createStudent(Map<String, dynamic> data) async {
       final response = await client.post('/students', data: data);
       return StaffStudentModel.fromJson(response.data);
     }

     Future<StaffStudentModel> updateStudent(String id, Map<String, dynamic> data) async {
       final response = await client.put('/students/$id', data: data);
       return StaffStudentModel.fromJson(response.data);
     }

     Future<void> deleteStudent(String id) async {
       await client.delete('/students/$id');
     }

     Future<void> toggleStatus(String id, bool activate) async {
       await client.put('/students/$id', data: {'status': activate ? 'active' : 'inactive'});
     }
   }
   ```

   ```dart
   import 'package:flutter_riverpod/flutter_riverpod.dart';
   import 'package:riverpod_annotation/riverpod_annotation.dart';
   import '../../../core/providers/dio_provider.dart';
   import '../../../core/providers/cache_provider.dart';
   import '../models/staff_student_model.dart';
   import '../services/staff_cadastro_service.dart';

   part 'staff_cadastro_provider.g.dart';

   @Riverpod(keepAlive: true)
   StaffCadastroService staffCadastroService(Ref ref) {
     final client = ref.watch(dioClientProvider);
     return StaffCadastroService(client: client);
   }

   @riverpod
   Future<List<StaffStudentModel>> staffStudents(Ref ref) async {
     final service = ref.watch(staffCadastroServiceProvider);
     final students = await service.getStudents();
     CacheTTL.schedule(ref, 'staffStudents');
     return students;
   }

   @riverpod
   class StaffCadastroFilter extends _$StaffCadastroFilter {
     @override
     String? build() => null; // null = "Todos"
     void setFilter(String? filter) => state = filter;
   }

   @riverpod
   class StaffCadastroSearch extends _$StaffCadastroSearch {
     @override
     String build() => '';
     void setQuery(String query) => state = query;
   }
   ```
